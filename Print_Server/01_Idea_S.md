    try:
        # –°–∫–∞—á–∏–≤–∞–µ–º —Ñ–∞–π–ª
        await bot.download_file(file_path, destination=str(destination))
        logging.info(f"–§–∞–π–ª '{file_name}' —É—Å–ø–µ—à–Ω–æ —Å–∫–∞—á–∞–Ω –≤ {destination}")

    except subprocess.CalledProcessError as e:
